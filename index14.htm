/* --- todas las funciones y constantes que compartiste --- */

/* Incluye loadAll, renderRes, renderCta, renderTN, renderTI, renderPr, renderLn, buildCfgList, renderCfgDetail, guardarCfgTodo, openPop, selCat, closePop, applyPop, fillCatSel, showTab, etc. */

 

/* 🔧 Ajuste dinámico de meses */

const hoy = new Date();

const MES_ACTUAL = hoy.getMonth() + 1; // Enero=1, Febrero=2, etc.

 

// Inicialmente, todos los meses anteriores al actual son reales/cerrados

let MESES_REALES = Array.from({length: MES_ACTUAL-1}, (_,i)=>i+1);

let MESES_CERRADOS = Array.from({length: MES_ACTUAL-1}, (_,i)=>i+1);

 

// Función para cerrar automáticamente un mes cuando se cargan sus cartolas

function marcarMesComoReal(mes){

  if(!MESES_REALES.includes(mes)) MESES_REALES.push(mes);

  if(!MESES_CERRADOS.includes(mes)) MESES_CERRADOS.push(mes);

}

 

/* 🔧 Ajuste en celdaCfg para permitir edición ilimitada en meses proyectados */

function celdaCfg(m,name,type){

  const lbl=MNS[(m-1)%12];

  const cerrado=MESES_CERRADOS.includes(m);

  const actual=(m===MES_ACTUAL);

  const esReal=MESES_REALES.includes(m);

  let celda='';

  if(type==='ing'){

    const v=getIng(m)[name]||0;

    if(cerrado){

      celda=`<div style="padding:4px;border:1px solid #b8ddb0;border-radius:3px;font-size:11px;text-align:right;background:#eaf7ec;color:#1a7a4a">$${fmt(v)}</div>`;

    } else {

      celda=`<input id="ci_${m}" type="number" value="${v}" style="width:100%;padding:4px;border:${actual?'2px solid #1a237e':'1px solid #ccc'};border-radius:3px;font-size:11px;text-align:right;background:${actual?'#f0f3ff':'#fff'}">`;

    }

  } else if(type==='grp'){

    const v=esReal?GRP[name].reduce((a,s)=>a+(getCostos(m)[s]||0),0):GRP[name].reduce((a,s)=>a+(getPpto(m)[s]||0),0);

    celda=`<div style="padding:4px;border:1px solid ${esReal?'#b8ddb0':'#eee'};border-radius:3px;font-size:11px;text-align:right;background:${esReal?'#eaf7ec':'#f5f5f5'};color:${esReal?'#1a7a4a':'#555'}">$${fmt(v)}</div>`;

  } else {

    if(cerrado){

      const vR=getCostos(m)[name]||0,vP=getPpto(m)[name]||0;

      celda=`<div title="Ppto: $${fmt(vP)}" style="padding:4px;border:1px solid #b8ddb0;border-radius:3px;font-size:11px;text-align:right;background:#eaf7ec;color:#1a7a4a;cursor:help">$${fmt(vR)}</div>`;

    } else {

      const vP=getPpto(m)[name]||0;

      celda=`<input id="cp_${m}" type="number" value="${vP}" style="width:100%;padding:4px;border:${actual?'2px solid #1a237e':'1px solid #ccc'};border-radius:3px;font-size:11px;text-align:right;background:${actual?'#f0f3ff':'#fff'}">`;

    }

  }

  return`<div style="text-align:center"><label style="font-size:9px;color:${cerrado?'#1a7a4a':actual?'#1a237e':'#888'};display:block;font-weight:${cerrado||actual?'700':'400'}">${lbl}${cerrado?' 🔒':actual?' ★':''}</label>${celda}</div>`;

}
